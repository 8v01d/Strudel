.decay(.04).sustain(0)
.gain(3)

$notespauses: note("c, eb, g")
.struct("x ~ x ~ ~ x ~ x ~ ~ ~ x ~ x ~ ~")
.slow(2)
.room(.6)
.size(.8)

$piano: "<0 [2 4]>"
.echoWith(4, 1/8, (p,n) => p.add(n*2))
.scale("C:ritusen").note()
.s("piano")
.room(.6)
.size(.8)
.shape(0.5)

.scale("C:ritusen")
